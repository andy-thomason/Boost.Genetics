DNA string Container class definitions. 


BOOST_GENETICS_UNROLL
namespace boost {
  namespace genetics {
    template<typename WordType, typename ArrayType> class basic_dna_string;

    typedef basic_dna_string< uint64_t, std::vector< uint64_t > > dna_string;
    typedef basic_dna_string< uint64_t, mapped_vector< uint64_t > > mapped_dna_string;
    template<> int get_code<dna_string >(const dna_string & str, size_t index);
    template<typename charT, typename traits, typename WordType, 
             typename Allocator> 
      std::basic_ostream< charT, traits > & 
      operator<<(std::basic_ostream< charT, traits > & os, 
                 const basic_dna_string< WordType, Allocator > & x);
    template<typename charT, typename traits, typename WordType, 
             typename Allocator> 
      std::basic_istream< charT, traits > & 
      operator>>(std::basic_istream< charT, traits > & is, 
                 basic_dna_string< WordType, Allocator > & x);
    template<typename WordType, typename ArrayType> 
      basic_dna_string< WordType, ArrayType > 
      rev_comp(const basic_dna_string< WordType, ArrayType > & x);
  }
}
