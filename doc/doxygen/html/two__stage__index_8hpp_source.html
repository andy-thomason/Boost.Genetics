147  void find_next(bool is_start) {
148  if (pos == dna_string::npos) {
149  // if we have already reached the end, stop.
150  return;
151  } else if (is_brute_force) {
152  size_t start = is_start ? pos : pos + 1;
153  if (start + search_str.size() > tsi->string->size()) {
154  pos = dna_string::npos;
155  } else {
156  pos = tsi->string->find_inexact(search_str, start, ~(size_t)0, max_distance);
157  }
158  return;
159  } else {
160  // for a small number of unknowns, use a merge to find potential starts.
161  addr_type prev_start = (addr_type)-1;
162  size_t repeat_count = 0;
163  pos = dna_string::npos;
164 
165  for (;;) {
166  merges_done++;
167  active_state s = active.front();
168 
169  if (s.start != prev_start) {
170  size_t erc = 0;
171  if (repeat_count >= num_seeds - max_distance) {
172  compares_done++;
173  if (tsi->string->compare_inexact(prev_start, search_str.size(), search_str, max_distance) == 0) {
174  pos = prev_start;
175  return;
176  }
177  }
178  repeat_count = 0;
179  prev_start = s.start;
180  }
181 
182  if (s.start == (addr_type)-1) {
183  //printf("merges_done=%lld\n", (long long)merges_done);
184  //printf("compares_done=%lld\n", (long long)compares_done);
185  return;
186  }
187 
188  // printf("xxx %5d %2d/%2d [%2d]\n", (int)s.start, (int)repeat_count, (int)(num_seeds - max_distance), (int)s.elem);
189 
190  const addr_type *ptr = s.ptr + 1;
191  const addr_type *end = s.end;
192  s.prev = s.start;
193  s.start = ptr == end ? (addr_type)-1 : (addr_type)(*ptr - s.elem * num_indexed_chars);
194  s.ptr = ptr;
195  std::pop_heap(active.begin(), active.end());
196  active.back() = s;
197  std::push_heap(active.begin(), active.end());
198 
199  // printf("xxx next: %d\n", (int)active.front().start);
200  repeat_count++;
201  }
202  }

284  // std::vector version has reindex()
285  template<>
286  void reindex_impl<true>() {
287  // todo: use threads to speed this up.
288  // consider using std::sort for better cache performance.
289  if (
290  num_indexed_chars <= 1 ||
291  num_indexed_chars > 32 ||
292  string->size() < num_indexed_chars ||
293  (addr_type)string->size() != string->size()
294  ) {
295  throw std::invalid_argument("two_stage_index::reindex()");
296  }
297 
298  size_t str_size = string->size();
299  size_t index_size = (size_t)1 << (num_indexed_chars*2);
300 
301  index.resize(0);
302  index.resize(index_size+1);
303  addr.resize(str_size);
304 
305  // lead-in: fill acc0 with first DNA codes
306  size_t acc0 = 0;
307  for (size_t i = 0; i != num_indexed_chars-1; ++i) {
308  acc0 = acc0 * 4 + get_code(*string, i);
309  }
310 
311  // count phase: count bucket sizes
312  size_t acc = acc0;
313  for (size_t i = num_indexed_chars; i != str_size; ++i) {
314  acc = (acc * 4 + get_code(*string, i)) & (index_size-1);
315  index[acc]++;
316  if (i % 0x100000 == 0) printf("c %5.2f\n", (double)i * (100.0/str_size));
317  }
318 
319  // compute running sum of buckets to convert counts to offsets.
320  addr_type cur = 0;
321  for (size_t i = 0; i != index_size; ++i) {
322  addr_type val = index[i];
323  index[i] = cur;
324  cur += val;
325  }
326  index[index_size] = cur;
327 
328  // store phase: fill "addr" with addresses of values.
329  acc = acc0;
330  for (size_t i = num_indexed_chars; i != str_size; ++i) {
331  acc = (acc * 4 + get_code(*string, i)) & (index_size-1);
332  addr[index[acc]++] = (addr_type)(i - num_indexed_chars + 1);
333  if (i % 0x100000 == 0) printf("s %5.2f\n", (double)i * (100.0/str_size));
334  }
335 
336  // shift the index up one so ends become starts.
337  addr_type prev = 0;
338  for (size_t i = 0; i != index_size; ++i) {
339  std::swap(prev, index[i]);
340  }
341  }
