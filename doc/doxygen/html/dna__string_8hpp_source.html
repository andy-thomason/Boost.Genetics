223  template <class String>
224  size_t find_inexact(
225  const String& search_str,
226  size_t pos = 0,
227  size_t n = ~(size_t)0,
228  size_t max_distance = 0
229  ) const {
230  size_t ssz = search_str.size();
231  if (ssz == 0) {
232  return pos;
233  }
234 
235  size_t sz = size();
236  if (pos >= sz) {
237  return basic_dna_string::npos;
238  }
239 
240  n = std::min((size_t)(sz - pos), n);
241  size_t last = pos + n;
242  if (pos + ssz > last || values.size() == 0) {
243  return basic_dna_string::npos;
244  }
245 
246  bool cpu_has_popcnt = has_popcnt();
247  bool cpu_has_lzcnt = has_lzcnt();
248  //printf("%d %d\n", cpu_has_popcnt, cpu_has_lzcnt);
249 
250  // Come gentle pedantry, shine upon this code.
251  const size_t bpv = bases_per_value;
252  size_t nv = std::min(values.size() - 1, last/bpv);
253  const word_type *search_values = search_str.get_values().data();
254  word_type s0 = search_values[0];
255  word_type s0mask = ssz >= bpv ? ~(word_type)0 : ~(word_type)0 << (bpv*2-ssz*2);
256  if (ssz >= 4 && max_distance == 0 && pos < nv * bpv) {
257  // Search for four characters, bpv at a time.
258  // This should be about 32 times faster than strstr.
259  word_type r1c = 0x5555555555555555ull;
260  word_type rep0 = ~((s0 >> (bpv*2-2)) * r1c);
261  word_type rep1 = ~(((s0 >> (bpv*2-4)) & 3) * r1c);
262  word_type rep2 = ~(((s0 >> (bpv*2-6)) & 3) * r1c);
263  word_type rep3 = ~(((s0 >> (bpv*2-8)) & 3) * r1c);
264  for (size_t i = pos/bpv; i < nv; ++i) {
265  word_type v0 = values[i];
266  word_type v1 = values[i+1];
267  word_type mask = v0 ^ rep0;
268  mask &= ((v0 << 2 ) | (v1 >> (bpv*2-2))) ^ rep1;
269  mask &= ((v0 << 4 ) | (v1 >> (bpv*2-4))) ^ rep2;
270  mask &= ((v0 << 6 ) | (v1 >> (bpv*2-6))) ^ rep3;
271  mask &= mask << 1;
272  mask &= 0xaaaaaaaaaaaaaaaaull;
273  size_t bit_pos = 0;
274  while (mask << bit_pos) {
275  int lz = (int)lzcnt(mask << bit_pos, cpu_has_lzcnt);
276  bit_pos += lz;
277  size_t search_pos = i * bpv + bit_pos/2;
278  word_type v = bit_pos == 0 ? v0 : (v0 << bit_pos) | (v1 >> (64-bit_pos));
279  ++bit_pos;
280  if (
281  ((v ^ s0) & s0mask) == 0 &&
282  search_pos >= pos && search_pos <= last - ssz &&
283  compare_inexact(search_pos, ssz, search_str, max_distance) == 0
284  ) {
285  return search_pos;
286  }
287  }
288  }
289 
290  pos = nv * bpv;
291  } else {
292  if (cpu_has_popcnt) {
293  pos = inexact_search<String, true>(search_str, pos, nv, s0, s0mask, max_distance, ssz, last);
294  } else {
295  pos = inexact_search<String, false>(search_str, pos, nv, s0, s0mask, max_distance, ssz, last);
296  }
297  if (pos != basic_dna_string::npos) {
298  return pos;
299  }
300  pos = nv * bpv;
301  }
302 
303  while (pos <= last - ssz) {
304  word_type w0 = window(pos);
305  if (count_word((w0 ^ s0) & s0mask, cpu_has_popcnt) <= max_distance) {
306  if (compare_inexact(pos, ssz, search_str, max_distance) == 0) {
307  return pos;
308  }
309  }
310  ++pos;
311  }
312  return basic_dna_string::npos;
313  }

352  template <class WordType, class ArrayType>
353  int compare_inexact(size_t pos, size_t ssz, const basic_dna_string<WordType, ArrayType> &str, size_t max_distance=0) const {
354  //printf("%s\n", std::string(str).c_str());
355  pos = std::min(pos, num_bases);
356  ssz = std::min(ssz, str.size());
357  ssz = std::min(ssz, num_bases - pos);
358 
359  bool cpu_has_popcnt = has_popcnt();
360  const ArrayType &str_values = str.get_values();
361  const size_t bpv = bases_per_value;
362  size_t nv = std::min(str_values.size(), (ssz+bpv-1)/bpv);
363  size_t error = 0;
364  for (size_t i = 0; i != nv; ++i) {
365  word_type w = window(pos);
366  word_type s = str_values[i];
367  if (i == ssz/bpv) {
368  s &= ~(word_type)0 << (((0-ssz) % bases_per_value) * 2);
369  w &= ~(word_type)0 << (((0-ssz) % bases_per_value) * 2);
370  }
371  if (s != w) {
372  if (max_distance == 0) {
373  return s == w ? 0 : s < w ? -1 : 1;
374  } else {
375  error += count_word(s^w, cpu_has_popcnt);
376  if (error > max_distance) {
377  return s < w ? -1 : 1;
378  }
379  }
380  }
381  pos += bpv;
382  }
383  return 0;
384  }
