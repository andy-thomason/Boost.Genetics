37  static inline bool is_base(unsigned int chr) {
38  static const uint64_t vals = (
39  (1 << ('A'-'A')) |
40  (1 << ('C'-'A')) |
41  (1 << ('G'-'A')) |
42  (1 << ('T'-'A'))
43  );
44  unsigned int idx = (chr & ~0x20) - 'A';
45  return idx <= 'T'-'A' && ((vals >> idx) & 1) != 0;
46  }

48  static inline int base_to_code(unsigned int chr) {
49  static const uint64_t vals = (
50  (1ull << ('C'-'C')*2) |

52  (3ull << ('T'-'C')*2)
53  );
54  unsigned int idx = (chr & ~0x20) - 'C';
55  return idx <= 'T'-'C' ? (int)(vals >> (idx*2)) & 0x03 : 0x00;

134  static inline int popcnt(uint64_t value, bool has_popcnt) {
135  #if defined(_MSC_VER) && defined(_M_X64)
136  if (has_popcnt) {
137  return (int)__popcnt64(value);
138  }
139  #endif
140  value = (value & 0x5555555555555555ull) + ((value >> 1) & 0x5555555555555555ull);
141  value = (value & 0x3333333333333333ull) + ((value >> 2) & 0x3333333333333333ull);
142  value = (value & 0x0f0f0f0f0f0f0f0full) + ((value >> 4) & 0x0f0f0f0f0f0f0f0full);
143  value = (value & 0x00ff00ff00ff00ffull) + ((value >> 8) & 0x00ff00ff00ff00ffull);
