103  void append(const char *p, const char *end) {
104  chromosome g;
105  g.start = 0;
106  size_t start = 0;
107  while (p != end) {
108  //if (*p != '>') throw(std::exception("bad fasta"));
109  const char *b = p;
110  while (p != end && *p != '\n' && *p != '\r') ++p;
111  size_t size = std::min(sizeof(g.info)-1, (size_t)(p-(b+1)));
112  memcpy(g.info, b+1, size);
113 
114  g.info[size] = 0;
115  size_t i = 0;
116  for (; i != size && i != sizeof(g.name)-1 && g.info[i] != ' '; ++i) {
117  g.name[i] = g.info[i];
118  }
119  g.name[i] = 0;
120 
121  b = p;
122  while (p != end && *p != '>') ++p;
123  str.append(b, p);
124  g.end = str.size();
125  chromosomes.push_back(g);
126  g.start = g.end;
127  }
