51  char operator[](size_t base) const {
52  size_t offset = base / bases_per_index;
53  if (offset > index.size()) return 'N';
54 
55  index_type i0 = index[offset];
56 
57  index_type i1 =
58  offset+1 >= index.size() ?
59  (index_type)rle.size() :
60  index[offset+1]
61  ;
62 
63  rle_type search = (rle_type)((base % bases_per_index) << 8) | 0xff;
64  const rle_type *b = rle.data() + i0;
65  const rle_type *e = rle.data() + i1;
66  const rle_type *p = std::lower_bound(b, e, search);
67  return p > b && (p[-1] & 0xff) ? (p[-1] & 0xff) : ((const parent&)*this)[base];

136  template<class InIter>
137  void internal_append(size_t num_bases, InIter b, InIter e) {
138  int prev_val = rle.empty() ? 0 : rle.back();
139  while (b != e) {
140  int chr = (int)*b++;
141  if (!is_whitespace(chr)) {
142  int val = (is_base(chr)) ? 0 : chr;
143  if (num_bases % bases_per_index == 0 || val != prev_val) {
144  if (num_bases % bases_per_index == 0) {
145  index.push_back((index_type)rle.size());
146  }
147  rle.push_back((rle_type)((num_bases % bases_per_index) * 256 + val));
148  prev_val = val;
149  }
150  num_bases++;
151  }
152  }
153  }
